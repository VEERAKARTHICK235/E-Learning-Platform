-- Create a new type for storing student info
CREATE TYPE student_type AS (
    name VARCHAR(50),
    age INT
);


-- Create a table that uses the new object type
CREATE TABLE students (
    id SERIAL,
    info student_type
);

-- Insert data into the table using object type
INSERT INTO students (info)
VALUES (ROW('Veera', 19)),
       (ROW('Karthick', 20));

-- Show all students
SELECT * FROM students;

-- Show only student names (access object field)
SELECT info.name FROM students;
